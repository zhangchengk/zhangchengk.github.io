   Map<Integer, LifecycleGroup> phases = new HashMap<>();

         int phase = getPhase(bean);
         LifecycleGroup group = phases.get(phase);
         if (group == null) {

         }

      Collections.sort(keys);
      for (Integer key : keys) {
         phases.get(key).start();
      }
   }
